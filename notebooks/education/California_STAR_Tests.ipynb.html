def plot_groups(subgroups, test_id, grade):
    df = p.select("""
SELECT * from star 
WHERE subgroup IN ({subgroups}) 
AND test_id = 7 -- CST English-Language Arts
AND school != 0
AND grade = grade
""".format(test_id=test_id,grade=grade,subgroups=','.join([str(x) for x in subgroups]))).pandas
    sdf = df[["subgroup","mean_score"]].dropna()
    
    colors = 'k','r','b','g','m','c'
    
    for i, sg in enumerate(subgroups):
        sdf[sdf['subgroup'] == sg]['mean_score'].plot(kind='kde', color=colors[i])
        #sdf[sdf['subgroup'] == sg]['mean_score'].hist(bins=100,color=colors[i])
